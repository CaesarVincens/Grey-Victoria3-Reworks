mog_plp_movement_anti_slavery_colonial_slavery_trigger = {
	AND = {
		Grey_PLP_is_active = yes # PLP - Colonial slavery is NOT out of sight and out of mind if you live in a colony
		state = { is_slave_state = no }
	}
}

mog_plp_movement_mainland_general_pop_support_trigger = {
	AND = {
		Grey_PLP_is_active = yes # PLP - Colonized pops don't care about mainland politics
		NOT = {
			AND = {
				state = { incorporation_progress = 0 }
				pop_has_primary_culture = no
			}
		}
	}
}
